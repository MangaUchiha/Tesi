% Gantt --------------------------------
\setganttlinklabel{f-s}{}
\setganttlinklabel{f-f}{}
\setganttlinklabel{s-s}{}

\newganttlinktype{ff}{
	\ganttsetstartanchor{on bottom=1}
	\ganttsetendanchor{on top = 1}
	\draw [/pgfgantt/link]
	(\xLeft, \yUpper) -- 
	(\xRight, \yUpper) --
	(\xRight, \yLower);
}

\newganttlinktype{fs}{
	\ganttsetstartanchor{on bottom=1}
	\ganttsetendanchor{on top = 0}
	\draw [/pgfgantt/link]
	(\xLeft, \yUpper) -- 
	(\xRight, \yUpper) --
	(\xRight, \yLower);
}

\newganttlinktype{fc}{
	\ganttsetstartanchor{on bottom=1}
	\ganttsetendanchor{on top = .5}

	(\xRight, \yUpper) --
	(\xRight, \yLower);
}
